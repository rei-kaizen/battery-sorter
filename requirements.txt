# ─────────────────────────────────────────────
#  HARDWARE REQUIREMENTS
# ─────────────────────────────────────────────

[microcontroller]
board               = NodeMCU ESP8266 (ESP-12E Module)
chip                = ESP8266
flash               = 4MB
usb_serial_chip     = CH340G
driver              = CH340 Windows Driver (wch-ic.com)
upload_mode         = auto-reset (no BOOT button needed)

[sensors]
ir_sensor           = FC-51 obstacle sensor, NPN output, active LOW, 5V
inductive_sensor    = NPN proximity sensor, 5–12V, active LOW on metal detect

[voltage_measurement]
method              = resistive voltage divider (production)
r1                  = 100kΩ (between battery+ and A0)
r2                  = 10kΩ  (between A0 and GND)
adc_pin             = A0 (NodeMCU)
adc_resolution      = 10-bit (0–1023)
input_range         = 0–3.3V (maps to 0–36.3V battery after 11x ratio)
formula             = V_battery = (raw / 1023.0) * 3.3 * 11.0
demo_mode           = simulatedVoltage() — random float 0.00–1.50V

[actuators]
servo_a             = SG90 or MG90S, 5V, signal GPIO 15 (D8) — Alkaline gate
servo_b             = SG90 or MG90S, 5V, signal GPIO  2 (D4) — NiMH gate
motor_driver        = L298N H-bridge module
conveyor_motor      = 12V DC motor

[display]
lcd                 = 16x2 LCD with PCF8574 I2C backpack
i2c_address         = 0x27 (try 0x3F if unresponsive)
sda_pin             = GPIO 4 (D2)
scl_pin             = GPIO 5 (D1)

[power]
motor_supply        = 12V DC adapter, min 2A (barrel jack)
logic_supply        = 5V USB power bank, min 1A
motor_wire_gauge    = 18 AWG (barrel jack → L298N VSS, L298N → motor)
signal_wire_gauge   = 22 AWG (all other connections)
wire_jacket         = silicone recommended for motor runs

[battery_types]
#  Type      Magnetic   Voltage Range   Servo Action
#  Li-ion    No         3.0 – 4.2V      Pass through (end of belt)
#  Alkaline  No         0.8 – 1.6V      Servo A → Bin A
#  NiMH      Yes        1.0 – 1.5V      Servo B → Bin B

[demo_mode]
voltage_source      = simulatedVoltage() random 0.00–1.50V
classification      = cyclic (Alkaline → NiMH → Li-ion → repeat)
toggle_to_prod      = replace simulatedVoltage() with ADC read in control.ino
